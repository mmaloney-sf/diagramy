////////////////////////////////////////
// Document
////////////////////////////////////////

    Document := Version Diagram BoxDef*


////////////////////////////////////////
// Diagram
////////////////////////////////////////

    Diagram := "diagram" "{" Prop* "}"


////////////////////////////////////////
// BoxDef
////////////////////////////////////////

    BoxDef := "box" Ident BoxBody

    BoxBody := BoxItem*

    BoxItem :=
          BoxInst
        | Prop
        | Port

    BoxInst :=
          "box" Ident? "at" Coord BoxBody
        | "box" Ident "at" Coord Ident

    Prop :=
          PropIdent
        | PropString
        | PropNumber
        | PropCoords

    PropIdent  := String ":" Ident
    PropString := String ":" String
    PropNumber := String ":" Num
    PropCoords := String ":" Coord

    Port :=
        "port" Ident


////////////////////////////////////////
// Extra
////////////////////////////////////////

    Version := "version" "=" VersionNumber

    VersionNumber := [0-9]+ "." [0-9]+ "." [0-9]+


    Coord := "(" Num "," Num ")"

////////////////////////////////////////
// Tokens
////////////////////////////////////////

    String := '"' [^"]* '"'

    Ident := [a-zA-Z_][a-zA-Z0-9_]*

    Num := [0-9]+
