////////////////////////////////////////
// Document
////////////////////////////////////////

    Document := Diagram BoxDef*


////////////////////////////////////////
// Diagram
////////////////////////////////////////

    Diagram := "diagram" "{" Prop* "}"


////////////////////////////////////////
// BoxDef
////////////////////////////////////////

    BoxDef := "box" Ident BoxBody

    BoxBody := "{" BoxItem* "}"

    BoxItem :=
          BoxInst
        | Prop
        | Port
        | Arrow

    BoxInst :=
          "box" Ident? ("at" Coords)? ("dim" Dim)? "is" BoxBody
        | "box" Ident? ("at" Coords)? ("dim" Dim)? "is" Ident

    Prop :=
          PropIdent
        | PropString
        | PropNumber
        | PropFrac
        | PropCoords
        | PropDim

////////////////////////////////////////
// Ports and Arrows
////////////////////////////////////////

    Port :=
        "port" Ident "at" Coords ("is""PortBody)?

    PortBody := "{" Prop* "}"

    Arrow :=
        "arrow" Ident "to" Ident ArrowBody?

    ArrowBody := "{" Prop* "}"


////////////////////////////////////////
// Properties
////////////////////////////////////////

    PropIdent  := Ident ":" Ident
    PropString := Ident ":" String+
    PropNumber := Ident ":" Num
    PropFrac   := Ident ":" Frac
    PropCoords := Ident ":" Coords
    PropDim    := Ident ":" Dim


////////////////////////////////////////
// Extra
////////////////////////////////////////

    Coords     := "(" Num "," Num ")"
    CoordsFrac := "(" Frac "," Frac ")"


////////////////////////////////////////
// Tokens
////////////////////////////////////////

    Ident  := [a-zA-Z_][a-zA-Z0-9_]*
    String := '"' [^"]* '"'
    Num    := [0-9]+
    Frac   := "-"? [0-9]+ "." [0-9]+
    Dim    := [0-9]+x[0-9]+
