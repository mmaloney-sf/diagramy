////////////////////////////////////////
// Document
////////////////////////////////////////

    Document := Version Diagram BoxDef*


////////////////////////////////////////
// Diagram
////////////////////////////////////////

    Diagram := "diagram" "{" Prop* "}"


////////////////////////////////////////
// BoxDef
////////////////////////////////////////

    BoxDef := "box" Ident BoxBody

    BoxBody := BoxItem*

    BoxItem :=
          BoxInst
        | Prop
        | Port

    BoxInst :=
          "box" Ident? "at" Coords Dim? BoxBody
        | "box" Ident? "at" Coords Dim? Ident

    Prop :=
          PropIdent
        | PropString
        | PropNumber
        | PropFrac
        | PropCoords
        | PropDim

    PropIdent  := Ident ":" Ident
    PropString := Ident ":" String+
    PropNumber := Ident ":" Num
    PropFrac   := Ident ":" Frac
    PropCoords := Ident ":" Coords
    PropDim    := Ident ":" Dim

    Port :=
        "port" Ident


////////////////////////////////////////
// Extra
////////////////////////////////////////

    Version := "version" "=" VersionNumber
    VersionNumber := [0-9]+ "." [0-9]+ "." [0-9]+
    Coords := "(" Num "," Num ")"


////////////////////////////////////////
// Tokens
////////////////////////////////////////

    Ident := [a-zA-Z_][a-zA-Z0-9_]*
    String := '"' [^"]* '"'
    Num := [0-9]+
    Frac := "-"? [0-9]+ ("." [0-9]+)?
    Dimensions := [0-9]+x[0-9]+
